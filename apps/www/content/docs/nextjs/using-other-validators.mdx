ArkEnv for Next.js supports any Standard Schema validator. That includes Zod, Valibot, Typia, and any validator listed here:

<Cards>
  <Card title="Which schema libraries implement the spec?" href="https://standardschema.dev/schema#what-schema-libraries-implement-the-spec" description="Standard Schema" />
</Cards>

### Flat layout

Here is an example using Zod or Valibot with the flat layout:

<Tabs items={['Zod', 'Valibot']}>
  <Tab value="Zod">
    ```ts title="src/env.ts" twoslash
    // @filename: generated/env.gen.ts
    import { createEnv as coreCreateEnv } from "@arkenv/nextjs";
    import type { type as at, distill } from "arktype";
    export { type } from "@arkenv/nextjs";
    export default function arkenv<const T extends {}>(
    	schema: T,
    	options?: any
    ): Readonly<distill.Out<at.infer<T>>> {
    	return {} as any;
    }

    // @filename: env.ts
    // ---cut---
    import arkenv from "./generated/env.gen";
    import { z } from "zod";

    export const env = arkenv({
    	DATABASE_URL: z.string().url(),
    	NEXT_PUBLIC_API_URL: z.string().url(),
    	NODE_ENV: z.enum(["development", "production", "test"]),
    });
    ```
  </Tab>

  <Tab value="Valibot">
    ```ts title="src/env.ts" twoslash
    // @filename: generated/env.gen.ts
    import { createEnv as coreCreateEnv } from "@arkenv/nextjs";
    import type { type as at, distill } from "arktype";
    export { type } from "@arkenv/nextjs";
    export default function arkenv<const T extends {}>(
    	schema: T,
    	options?: any
    ): Readonly<distill.Out<at.infer<T>>> {
    	return {} as any;
    }

    // @filename: env.ts
    // ---cut---
    import arkenv from "./generated/env.gen";
    import * as v from "valibot";

    export const env = arkenv({
    	DATABASE_URL: v.pipe(v.string(), v.url()),
    	NEXT_PUBLIC_API_URL: v.pipe(v.string(), v.url()),
    	NODE_ENV: v.picklist(["development", "production", "test"]),
    });
    ```
  </Tab>
</Tabs>

### Strict layout

You can freely mix Zod or Valibot schemas with the strict layout:

<Tabs items={['Zod', 'Valibot']}>
  <Tab value="Zod">
    ```ts title="src/env/internal/shared.ts" twoslash
    import { z } from "zod";

    /**
     * @internal 🛑 INTERNAL SCHEMA ONLY.
     * Do not import this directly. Import `env` from `./client` or `./server` instead.
     */
    export const SharedSchema = z.object({
    	NODE_ENV: z.enum(["development", "production", "test"]).default("development"),
    });
    ```
  </Tab>

  <Tab value="Valibot">
    ```ts title="src/env/internal/shared.ts" twoslash
    import * as v from "valibot";

    /**
     * @internal 🛑 INTERNAL SCHEMA ONLY.
     * Do not import this directly. Import `env` from `./client` or `./server` instead.
     */
    export const SharedSchema = v.object({
    	NODE_ENV: v.optional(v.picklist(["development", "production", "test"]), "development"),
    });
    ```
  </Tab>
</Tabs>
